    import asyncio
from collections import deque
from cryptoservice.models import Freq

class RealTimeAnalyzer:
    def __init__(self, symbol="BTCUSDT", window_size=20):
        self.ws_service = WebSocketService()
        self.symbol = symbol
        self.window_size = window_size
        self.price_buffer = deque(maxlen=window_size)
        self.volume_buffer = deque(maxlen=window_size)

    async def start_analysis(self):
        """ÂêØÂä®ÂÆûÊó∂ÂàÜÊûê"""
        await self.ws_service.connect()
        await self.ws_service.subscribe_kline(self.symbol, Freq.m1)

        print(f"üìä {self.symbol} ÂÆûÊó∂ÊäÄÊúØÂàÜÊûêÂêØÂä®...")

        async for data in self.ws_service.listen():
            if data.type == "kline" and data.is_closed:
                await self.analyze_kline(data)

    async def analyze_kline(self, kline_data):
        """ÂàÜÊûêKÁ∫øÊï∞ÊçÆ"""
        close_price = float(kline_data.close_price)
        volume = float(kline_data.volume)

        # Êõ¥Êñ∞ÁºìÂÜ≤Âå∫
        self.price_buffer.append(close_price)
        self.volume_buffer.append(volume)

        if len(self.price_buffer) >= self.window_size:
            # ËÆ°ÁÆóÊäÄÊúØÊåáÊ†á
            sma = sum(self.price_buffer) / len(self.price_buffer)

            # ËÆ°ÁÆóÊ≥¢Âä®Áéá
            price_changes = [
                (self.price_buffer[i] - self.price_buffer[i-1]) / self.price_buffer[i-1]
                for i in range(1, len(self.price_buffer))
            ]
            volatility = (sum(x**2 for x in price_changes) / len(price_changes)) ** 0.5 * 100

            # Êàê‰∫§ÈáèÂàÜÊûê
            avg_volume = sum(self.volume_buffer) / len(self.volume_buffer)
            volume_ratio = volume / avg_volume

            # Ë∂ãÂäøÂà§Êñ≠
            recent_prices = list(self.price_buffer)[-5:]
            trend = "‰∏äÊ∂®" if recent_prices[-1] > recent_prices[0] else "‰∏ãË∑å"

            # ËæìÂá∫ÂàÜÊûêÁªìÊûú
            print(f"\nüìä {self.symbol} ÊäÄÊúØÂàÜÊûê ({datetime.now().strftime('%H:%M:%S')})")
            print(f"ÂΩìÂâç‰ª∑Ê†º: ${close_price:,.2f}")
            print(f"ÁßªÂä®Âπ≥Âùá: ${sma:,.2f}")
            print(f"Ê≥¢Âä®Áéá: {volatility:.2f}%")
            print(f"Êàê‰∫§ÈáèÂÄçÊï∞: {volume_ratio:.2f}x")
            print(f"Áü≠ÊúüË∂ãÂäø: {trend}")

            # ‰ø°Âè∑Ê£ÄÊµã
            await self.detect_signals(close_price, sma, volume_ratio)

    async def detect_signals(self, price, sma, volume_ratio):
        """Ê£ÄÊµã‰∫§Êòì‰ø°Âè∑"""
        signals = []

        # ‰ª∑Ê†ºÁ™ÅÁ†¥‰ø°Âè∑
        if price > sma * 1.02:  # ‰ª∑Ê†ºÁ™ÅÁ†¥ÁßªÂä®Âπ≥Âùá2%
            signals.append("üî∂ ‰ª∑Ê†ºÁ™ÅÁ†¥ÁßªÂä®Âπ≥ÂùáÁ∫ø")

        # Êàê‰∫§ÈáèÂºÇÂ∏∏‰ø°Âè∑
        if volume_ratio > 2.0:  # Êàê‰∫§ÈáèÊòØÂπ≥ÂùáÂÄºÁöÑ2ÂÄç‰ª•‰∏ä
            signals.append("üìà Êàê‰∫§ÈáèÊîæÂ§ß")

        # ËæìÂá∫‰ø°Âè∑
        if signals:
            print("üîî ‰∫§Êòì‰ø°Âè∑:")
            for signal in signals:
                print(f"   {signal}")

    def scan_market_hotspots():

            try:
                analysis_data.append({
                    'symbol': ticker.symbol,
                    'price': float(ticker.last_price),
                    'change_percent': float(ticker.price_change_percent),
                    'volume': float(ticker.volume),
                    'quote_volume': float(ticker.quote_volume),
                    'count': int(ticker.count)

                continue

    # ÊéíÂ∫èÂíåÁ≠õÈÄâ

    print("\nüî• Ê∂®ÂπÖÊ¶ú TOP 10:")

    print("\nüìâ Ë∑åÂπÖÊ¶ú TOP 10:")

    for _, row in top_losers.iterrows():

    print("\nüí∞ Êàê‰∫§È¢ùÊ¶ú TOP 10:")
    top_volume = df.nlargest(10, 'quote_volume')
    for _, row in top_volume.iterrows():
        volume_millions = row['quote_volume'] / 1_000_000
        print(f"üí∞ {row['symbol']}: ${volume_millions:.1f}M")

    return df

    import numpy as np
import seaborn as sns

def correlation_analysis():
    """Âä†ÂØÜË¥ßÂ∏ÅÁõ∏ÂÖ≥ÊÄßÂàÜÊûê"""

    # ‰∏ªË¶ÅÂä†ÂØÜË¥ßÂ∏Å
    major_cryptos = ["BTCUSDT", "ETHUSDT", "BNBUSDT", "ADAUSDT", "SOLUSDT", "DOTUSDT"]

    # Ëé∑ÂèñÂéÜÂè≤Êï∞ÊçÆ
    end_time = datetime.now()
    start_time = end_time - timedelta(days=30)

    price_data = {}
    for symbol in major_cryptos:
        klines = service.get_historical_klines(
            symbol=symbol,
            start_time=start_time.strftime("%Y-%m-%d"),
            end_time=end_time.strftime("%Y-%m-%d"),
            interval=Freq.d1
        )

        prices = [float(k.close_price) for k in klines]
        price_data[symbol.replace("USDT", "")] = prices

    # ÂàõÂª∫DataFrame
    df = pd.DataFrame(price_data)

    # ËÆ°ÁÆóÊî∂ÁõäÁéá
    returns = df.pct_change().dropna()

    # ËÆ°ÁÆóÁõ∏ÂÖ≥ÊÄßÁü©Èòµ
    correlation_matrix = returns.corr()

    print("üìä Âä†ÂØÜË¥ßÂ∏ÅÁõ∏ÂÖ≥ÊÄßÂàÜÊûê (ËøáÂéª30Â§©)")
    print("=" * 50)
    print(correlation_matrix.round(3))

    # ÂèØËßÜÂåñÁõ∏ÂÖ≥ÊÄßÁÉ≠Âõæ
    plt.figure(figsize=(10, 8))
    sns.heatmap(
        correlation_matrix,
        annot=True,
        cmap='RdYlBu_r',
        center=0,
        square=True,
        fmt='.3f'
    )
    plt.title('Âä†ÂØÜË¥ßÂ∏Å‰ª∑Ê†ºÁõ∏ÂÖ≥ÊÄßÁÉ≠Âõæ')
    plt.tight_layout()
    plt.show()

    # ÂØªÊâæÊúÄÁõ∏ÂÖ≥ÂíåÊúÄ‰∏çÁõ∏ÂÖ≥ÁöÑÂ∏ÅÂØπ
    correlations = []
    for i in range(len(correlation_matrix.columns)):
        for j in range(i+1, len(correlation_matrix.columns)):
            correlations.append({
                'pair': f"{correlation_matrix.columns[i]}-{correlation_matrix.columns[j]}",
                'correlation': correlation_matrix.iloc[i, j]
            })

    correlations.sort(key=lambda x: abs(x['correlation']), reverse=True)

    print("\nüîó ÊúÄÁõ∏ÂÖ≥ÁöÑÂ∏ÅÂØπ:")
    for item in correlations[:3]:
        print(f"   {item['pair']}: {item['correlation']:.3f}")

    print("\nüîÄ ÊúÄ‰∏çÁõ∏ÂÖ≥ÁöÑÂ∏ÅÂØπ:")
    for item in correlations[-3:]:
        print(f"   {item['pair']}: {item['correlation']:.3f}")

class PriceAlertSystem:
    def __init__(self):

    def add_alert(self, symbol, condition, target_price, message=""):
        """Ê∑ªÂä†‰ª∑Ê†ºË≠¶Êä•"""
        if symbol not in self.alerts:
            self.alerts[symbol] = []

        self.alerts[symbol].append({
            'condition': condition,  # 'above' or 'below'
            'target_price': target_price,
            'message': message,
            'triggered': False
        })

        print(f"‚úÖ Ë≠¶Êä•Â∑≤Ê∑ªÂä†: {symbol} {condition} ${target_price}")

    async def start_monitoring(self):
        """ÂêØÂä®Ë≠¶Êä•ÁõëÊéß"""
        await self.ws_service.connect()

        # ËÆ¢ÈòÖÊâÄÊúâÈúÄË¶ÅÁõëÊéßÁöÑ‰∫§ÊòìÂØπ
        for symbol in self.alerts.keys():
            await self.ws_service.subscribe_ticker(symbol)

        print("üö® ‰ª∑Ê†ºË≠¶Êä•Á≥ªÁªüÂêØÂä®...")

        async for data in self.ws_service.listen():
            await self.check_alerts(data)

    async def check_alerts(self, ticker_data):
        """Ê£ÄÊü•Ë≠¶Êä•Êù°‰ª∂"""
        symbol = ticker_data.symbol
        current_price = float(ticker_data.last_price)

        if symbol in self.alerts:
            for alert in self.alerts[symbol]:
                if not alert['triggered']:
                    condition_met = False

                    if alert['condition'] == 'above' and current_price >= alert['target_price']:
                        condition_met = True
                    elif alert['condition'] == 'below' and current_price <= alert['target_price']:
                        condition_met = True

                    if condition_met:
                        await self.trigger_alert(symbol, current_price, alert)
                        alert['triggered'] = True

    async def trigger_alert(self, symbol, current_price, alert):
        """Ëß¶ÂèëË≠¶Êä•"""
        message = f"üö® ‰ª∑Ê†ºË≠¶Êä•Ëß¶Âèë!\n"
        message += f"‰∫§ÊòìÂØπ: {symbol}\n"
        message += f"ÂΩìÂâç‰ª∑Ê†º: ${current_price:,.2f}\n"
        message += f"ÁõÆÊ†á‰ª∑Ê†º: ${alert['target_price']:,.2f}\n"
        message += f"Êù°‰ª∂: {alert['condition']}\n"

        if alert['message']:
            message += f"Â§áÊ≥®: {alert['message']}\n"

        print(message)

        # ÂèëÈÄÅÈÇÆ‰ª∂ÈÄöÁü•
        await self.send_email_alert(symbol, message)

    async def send_email_alert(self, symbol, message):
        """ÂèëÈÄÅÈÇÆ‰ª∂Ë≠¶Êä•"""
        try:
            msg = MimeMultipart()
            msg['From'] = self.email_config['email']
            msg['To'] = self.email_config['email']
            msg['Subject'] = f"CryptoService ‰ª∑Ê†ºË≠¶Êä•: {symbol}"

            msg.attach(MimeText(message, 'plain'))

            server = smtplib.SMTP(self.email_config['smtp_server'], self.email_config['smtp_port'])
            server.starttls()
            server.login(self.email_config['email'], self.email_config['password'])
            server.send_message(msg)
            server.quit()

            print("üìß ÈÇÆ‰ª∂Ë≠¶Êä•Â∑≤ÂèëÈÄÅ")

        except Exception as e:
            print(f"‚ùå ÈÇÆ‰ª∂ÂèëÈÄÅÂ§±Ë¥•: {e}")
